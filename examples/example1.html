PHP 5.4.7

Example 1

1. Construct

Tag::a()

<a></a>

Tag::create('hoge', array('class' => 'fuga'), 'munya')

<hoge class="fuga"> munya </hoge>

Tag::div(array('class' => 'span12'))

<div class="span12"> </div>

Tag::div(array('class' => 'span12'), 'test', Tag::span('inner'))

<div class="span12"> test <span>inner</span> </div>

2. Member

Tag::table()->tagName()

table

Tag::table(array('class' => 'horizontal'))->attributes()

array(1) { ["class"]=> string(10) "horizontal" }

3. Attributes

Tag::div()->class('control-group')

<div class="control-group"> </div>

Tag::a('here')->href('http://www.google.com')

<a href="http://www.google.com">here</a>

Tag::a('another')->attr('href', 'http://www.yahoo.com')

<a href="http://www.yahoo.com">another</a>

4. Append

Tag::div()->append(Tag::span('content'))

<div> <span>content</span> </div>

5. Doctype

Tag::$DocType = 'html4'; Tag::br()

<br>

Tag::$DocType = 'xhtml'; Tag::br()

<br />

Tag::$DocType = 'html5'; Tag::br()

<br>
